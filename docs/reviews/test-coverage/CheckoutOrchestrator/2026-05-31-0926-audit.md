public async Task MultipleConcurrentCheckouts_WithSameIdempotencyKey_OnlyCreatesOneSaga()
{
    // Arrange
    var idempotencyKey = "concurrent-test";
    var tasks = Enumerable.Range(0, 5).Select(_ => 
        StartCheckoutAsync(idempotencyKey));

    // Act
    var results = await Task.WhenAll(tasks);

    // Assert  
    var sagaIds = results.Select(r => r.SagaId).Distinct();
    sagaIds.Should().HaveCount(1, "Same idempotency key should produce same saga ID");
}
