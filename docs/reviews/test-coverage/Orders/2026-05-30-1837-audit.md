[Collection("Orders Integration")]
public class PrivacyErasureConsumerIntegrationTests : IAsyncLifetime
{
    private readonly OrdersWebAppFactory _factory;
    private readonly HttpClient _client;

    public PrivacyErasureConsumerIntegrationTests(OrdersWebAppFactory factory)
    {
        _factory = factory;
        _client = factory.CreateClient();
    }

    [Fact]
    public async Task Consume_ErasureRequest_AnonymisesUserOrders()
    {
        // Arrange - Create orders for user
        var userId = Guid.NewGuid();
        var (orderId1, _) = await CreateOrderAsync(userId.ToString());
        var (orderId2, _) = await CreateOrderAsync(userId.ToString());

        var harness = _factory.Services.GetRequiredService<ITestHarness>();

        // Act
        await harness.Bus.Publish(new PrivacyErasureRequested
        {
            UserId = userId,
            RequestId = Guid.NewGuid()
        });

        // Assert - Poll for completion event
        await TestWait.Until(async () =>
        {
            return harness.Published.Select<PrivacyErasureCompleted>()
                .Any(p => p.Context.Message.UserId == userId);
        }, TimeSpan.FromSeconds(30));

        // Verify anonymisation
        using var scope = _factory.Services.CreateScope();
        var db = scope.ServiceProvider.GetRequiredService<OrderDbContext>();
        var anonymisedOrders = await db.Orders.AsNoTracking()
            .Where(o => o.Id == orderId1 || o.Id == orderId2)
            .ToListAsync();

        anonymisedOrders.Should().AllSatisfy(o =>
        {
            o.UserId.Should().Be("ERASED");
            o.CustomerEmail.Should().Be("deleted@privacy.invalid");
        });
    }
}
